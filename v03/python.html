df.groupby('region')['sales'].mean()

# merge — SQL-style join

NumPy Patterns

Common

Vectorized operations are the key to fast numerical computing. Skip the for-loops.

import numpy as np

a = np.array([1, 2, 3, 4, 5])
np.mean(a)                  # 3.0
np.dot(a, a)                # 55

# linear algebra
M = np.array([[1,2],[3,4]])
np.linalg.inv(M)           # matrix inverse
np.linalg.eig(M)           # eigenvalues

# boolean indexing
a[a > 3]                    # array([4, 5])
M[M % 2 == 0]              # array([2, 4])

Common Coding Patterns

Rare

Patterns that show up again and again in DS coding rounds. Know them cold.

# Two pointers
def two_sum_sorted(nums, target):
    l, r = 0, len(nums) - 1
    while l < r:
        s = nums[l] + nums[r]
        if s == target: return [l, r]
        elif s < target: l += 1
        else: r -= 1

# Hash map for O(1) lookup
seen = {}
for i, v in enumerate(nums):
    if target - v in seen:
        return [seen[target - v], i]
    seen[v] = i

# Sliding window
from collections import Counter
window = Counter(s[:k])
for i in range(k, len(s)):
    window[s[i]] += 1
    window[s[i - k]] -= 1

String & List Tricks

Common

Pythonic one-liners that speed up your coding and impress interviewers.

# List comprehension
squares = [x**2 for x in range(10) if x % 2 == 0]

# zip — iterate in parallel
for name, score in zip(names, scores):
    print(f"{name}: {score}")

# sorted with key
sorted(students, key=lambda s: s[1], reverse=True)

# dict comprehension
freq = {ch: s.count(ch) for ch in set(s)}

# unpacking
first, *middle, last = [1, 2, 3, 4, 5]

Gotchas

Legendary

These traps have ended many interview runs. Study them or perish.

# Mutable default argument — DANGER
def bad(x=[]):       # list is shared across calls!
    x.append(1)
    return x
bad()                  # [1]
bad()                  # [1, 1]  <-- not [1]!

# Shallow copy trap
a = [[1, 2], [3, 4]]
b = a.copy()           # shallow! inner lists shared
b[0].append(99)
print(a[0])            # [1, 2, 99] — surprise!

# Integer division
7 / 2                   # 3.5  (true division)
7 // 2                  # 3    (floor division)
-7 // 2                 # -4   (floors toward -inf!)

# is vs ==
a = [1, 2]; b = [1, 2]
a == b                  # True  (equal values)
a is b                  # False (different objects)
